#!/bin/bash
sudo yum install git -y 

su - ec2-user <<EOF
echo "User home $HOME"
cd $HOME
curl -o- https://raw.githubusercontent.com/creationix/nvm/v0.33.6/install.sh | bash 
source .bashrc
nvm install --lts 
npm install pm2 -g
git clone -b version/aws-ec2-pm2 https://github.com/jrierapeiro/node_web_app.git
cd node_web_app
npm install
npm run restart

EOF
